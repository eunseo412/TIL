### 입출력 함수, 변수
---
`#include <iostream> 
`#` : 전처리기. 컴파일을 시작하면 우선적으로 처리
// iostream: c++의 표준 입출력 함수가 정의되어있는 헤더파일
// input output stream의 약자
int main()
{
    std::cout << "Hello world!"
    int a = 3;
    int b;
    b = 10;
    // 자료형을 정해서 변수를 넣을 공간을 만든 후 데이터를 넣어야 함
    std::cin >> a >> b;
    // console input 약자
    // 입력받은 값을 변수(박스)안에 넣어주는 명령어
    std::cout << a << b << std::endl;
    // 변수에 있는 값을 증가시키고 싶을 때
    a = a + 5;
    a += 5;
    a++;
    a--;
    // 1을 더하거나 빼기
    b = ++a; 
    // 전위연산자
    b = a++; 
    // 후위연산자
    return 0;`
